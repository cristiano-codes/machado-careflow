  839: function normalizeWeeklyScale(
  840:   value,
  841:   defaultWeekScale = DEFAULT_WEEK_SCALE,
  842:   defaultStartTime = DEFAULT_DAY_START_TIME,
  843:   defaultEndTime = DEFAULT_DAY_END_TIME
  844: ) {
  845:   const startDefaultMinutes = parseTimeToMinutes(defaultStartTime);
  846:   const endDefaultMinutes = parseTimeToMinutes(defaultEndTime);
  847: 
  848:   const safeDefaultStart = startDefaultMinutes === null ? DEFAULT_DAY_START_TIME : defaultStartTime;
  849:   const safeDefaultEnd = endDefaultMinutes === null ? DEFAULT_DAY_END_TIME : defaultEndTime;
  850: 
  851:   if (value === undefined || value === null || value === '') {
  852:     const fallback = {};
  853:     for (const key of SCALE_KEYS) {
  854:       fallback[key] = {
  855:         ativo: Boolean(defaultWeekScale[key]),
  856:         inicio: safeDefaultStart,
  857:         fim: safeDefaultEnd,
  858:       };
  859:     }
  860:     return fallback;
  861:   }
  862: 
  863:   let raw = value;
  864:   if (typeof raw === 'string') {
  865:     try {
  866:       raw = JSON.parse(raw);
  867:     } catch (error) {
  868:       return null;
  869:     }
  870:   }
  871: 
  872:   if (!raw || typeof raw !== 'object' || Array.isArray(raw)) {
  873:     return null;
  874:   }
  875: 
  876:   const normalized = {};
  877:   for (const key of SCALE_KEYS) {
  878:     const defaultDay = {
  879:       ativo: Boolean(defaultWeekScale[key]),
  880:       inicio: safeDefaultStart,
  881:       fim: safeDefaultEnd,
  882:     };
  883: 
  884:     const current = raw[key];
  885:     if (current === undefined) {
  886:       normalized[key] = defaultDay;
  887:       continue;
  888:     }
  889: 
  890:     if (typeof current === 'boolean') {
  891:       normalized[key] = {
  892:         ...defaultDay,
  893:         ativo: current,
  894:       };
  895:       continue;
  896:     }
  897: 
  898:     if (!current || typeof current !== 'object' || Array.isArray(current)) {
  899:       return null;
  900:     }
  901: 
  902:     const activeValue =
  903:       current.ativo !== undefined
  904:         ? current.ativo
  905:         : current.active !== undefined
  906:           ? current.active
  907:           : current.enabled !== undefined
  908:             ? current.enabled
  909:             : undefined;
  910: 
  911:     if (activeValue !== undefined && typeof activeValue !== 'boolean') {
  912:       return null;
  913:     }
  914: 
  915:     const inicioValue =
  916:       typeof current.inicio === 'string'
  917:         ? current.inicio.trim()
  918:         : typeof current.start === 'string'
  919:           ? current.start.trim()
  920:           : defaultDay.inicio;
  921: 
  922:     const fimValue =
  923:       typeof current.fim === 'string'
  924:         ? current.fim.trim()
  925:         : typeof current.end === 'string'
  926:           ? current.end.trim()
  927:           : defaultDay.fim;
  928: 
  929:     const inicioMinutes = parseTimeToMinutes(inicioValue);
  930:     const fimMinutes = parseTimeToMinutes(fimValue);
  931:     if (inicioMinutes === null || fimMinutes === null) {
  932:       return null;
  933:     }
  934: 
  935:     const ativo = typeof activeValue === 'boolean' ? activeValue : defaultDay.ativo;
  936:     if (ativo && inicioMinutes >= fimMinutes) {
  937:       return null;
  938:     }
  939: 
  940:     normalized[key] = {
  941:       ativo,
  942:       inicio: inicioValue,
  943:       fim: fimValue,
  944:     };
  945:   }
  946: 
  947:   return normalized;
  948: }
